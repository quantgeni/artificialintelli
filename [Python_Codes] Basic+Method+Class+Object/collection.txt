import collections


# 리스트 내 각 요소의 빈도를 파악하세요
val = [1,3,2,1,1,2,3,4,2,1,1,1,2]

print(collections.Counter(val))


# 문자열에도 Counter 함수 적용하기
str1 = 'aabcdebagffde'

print(collections.Counter(str1))


text = "I am honored to be with you today at your commencement from one of the finest universities in the world. I never graduated from college. Truth be told, this is the closest I've ever gotten to a college graduation. Today I want to tell you three stories from my life. That's it. No big deal. Just three stories."
words = text.split()
# 문장을 공백을 기준으로 각 단어로 잘라냄

# print(collections.Counter(words))

wdcnt = collections.Counter(words)

print(wdcnt.get('I', 0))
# I 단어의 빈도수 출력, 없으면 0으로 출력

print(wdcnt.get('xyz', 0))



# 최상위 빈도수 출력 : most_commons
chars = collections.Counter(val)
# print(chars)
print(chars.most_common(3))

chars = collections.Counter(str1)
# print(chars)
print(chars.most_common(5))
